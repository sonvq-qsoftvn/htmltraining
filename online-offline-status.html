Check Online/Offline Status example - Team 5 - Offline Storage

Display online/offline status by window.addEventListener

Display online/offline status by navigator.onLine

Code used in this page

(function () {
	var imgListener = document.getElementById("img-listener"),
		imgNavigator = document.getElementById("img-navigator"),
		isOnline = function () {
			imgListener.src = "yahoo-on.png";
		},
		isOffline = function () {
			imgListener.src = "yahoo-off.png";
		};

		window.addEventListener("online", isOnline, false);
		window.addEventListener("offline", isOffline, false);
		
		setInterval(
			function(){ 
				if (navigator.onLine) { 
					imgNavigator.src = "skype-on.png"; 
				} else { 
					imgNavigator.src = "skype-off.png"; 
				}
			}, 500);
		
		
})();
